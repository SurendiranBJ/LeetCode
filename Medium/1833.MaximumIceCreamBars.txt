class Solution(object):
    def maxIceCream(self, costs, coins):
        """
        :type costs: List[int]
        :type coins: int
        :rtype: int
        """
        ice=0
        costs.sort()
        for i in range(len(costs)):
            if costs[i]<=coins:
                ice+=1
                coins-=costs[i]
        return ice
